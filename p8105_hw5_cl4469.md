t_test = function(mu, n = 30, sigma = 5) {
  
  sim_data = tibble(
    x = rnorm(n, mean = mu, sd = sigma)
    )
  
  sim_result = broom::tidy(t.test(sim_data))
  sim_result |> 
    dplyr::select(estimate, p.value)
}
